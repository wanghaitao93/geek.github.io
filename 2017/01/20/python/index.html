parse code

import ConfigParser
def read_config_trans_data():
    """
    read config file trans dict
    """
    print 'start read config file....................'
    config_dict = {}

    cf = ConfigParser.ConfigParser()
    cf.read('conf.conf')
    secs = cf.sections()

    for sec in secs:

        opts = cf.options(sec)
        opt_dict = {}

        for opt in opts:
            opt_dict[opt] = cf.get(sec, opt).strip()
            print '%s = %s' % (opt, cf.get(sec, opt).strip())

        config_dict[sec] = opt_dict

    print 'end.....................................'
    return config_dict

os

访问操作系统的服务功能

os.walk遍历文件夹的文件

for root, dirs, files in os.walk(input_file_path):
for f in files:
    current_file_path = os.path.join(root, f)

os.system & os.popen
os.system 返回执行shell命令后的状态码，成功是0，失败是-1
os.popen 返回执行shell命令后的输出信息。

os.path
os.path.isfile(file) 如果文件不存在返回False
os.path.exists(directory) 如果目录不存在返回False

from lxml import etree

parser = etree.XMLParser(strip_cdata=False)
self.schema_xml = etree.parse("./schema.xml", parser = parser)
self.g_item = etree.parse("./schema.xml", parser = parser).find("item")
g_item = self.schema_xml.find("item")
self.schema_xml.getroot().remove(g_item)

# 删除当前的xml
new_item.find("display/list").getparent().remove(new_item.find("display/list"))

# 去除获取xml的空格
parse = etree.XMLParser(remove_blank_text = True)
xml = etree.parse(current_file_path, parse)


# 添加新的元素
img_element = product_item.find(img)
