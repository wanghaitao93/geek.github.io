python multi process & thread

Posted on 2017-01-30 | In python | |

使用python开发不可能不会多进程和多线程编程技巧。
以下是我使用python的多线程和多进程demo。

多线程

import threading
import time


count = 0
count_lock = threading.Lock()

class MyThread(threading.Thread):
    def __init__(self, name):
        threading.Thread.__init__(self)
        self.name = "Thread-" + str(name)

    def run(self):
        global count, count_lock

        while True:

            if count_lock.acquire():
                count += 1
                print "Im'a %s, set counter: %s " % (self.name, count)
                count_lock.release()


if __name__ == "__main__":

    threadList = []

    for i in xrange(1, 10):
        my_thread = MyThread(i)
        threadList.append(my_thread)
    for t in threadList:
        t.start()
    for t in threadList:
        t.join()
