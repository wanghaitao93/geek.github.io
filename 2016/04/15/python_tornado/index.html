使用python作为服务器，使用过httpserver，但是并发量一大就会有丢数据的现象，tornado大大提高了并发接受http请求的数量。
以下是我写的tornado的demo。

Tornado demo

源代码
#!/usr/bin/env python
# -*- coding: utf-8 -*-

from tornado import httpserver
from tornado import escape
from tornado import ioloop
from tornado import web
import logging, subprocess

logging.basicConfig(level = logging.INFO)

class MainHandler(tornado.web.RequestHandler):
        """
        inheritance tornado.web.RequestHandler, override get and post method
        """
    def get(self, data):
        self.write("get, Hello, world")
        request = self.request
        logging.info('get self.response : %s', request)
        logging.info('get data : %s', data)

    def post(self, data):
        self.write("post, Hello, world")  
        request = self.request
        logging.info('post self.response : %s', request)
        logging.info('post data : %s', data)
        logging.info('post self.response.body : %s', self.request.body)

def make_app():
    return tornado.web.Application([
        (r"/(.*)", MainHandler),
    ])

def openServer():
    app = make_app()
    sockets = tornado.netutil.bind_sockets(8888)
    # 最大限度开启子线程数
    tornado.process.fork_processes(0)   
    server = httpserver.HTTPServer(app)
    server.add_sockets(sockets)
    ioloop.IOLoop.current().start()
    logging.info('server listen 8888...')

openServer()
